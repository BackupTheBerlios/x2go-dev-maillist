Using:

#!/usr/bin/python
import paramiko

ssh = paramiko.SSHClient()
ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
ssh.connect(
    "weblive-appserv01.nx.stgraber.org",
    port=6522,
    username="root",
    password="MY PASSWORD",
    allow_agent=False,
    look_for_keys=False,
    timeout=10
)
ssh.close()
