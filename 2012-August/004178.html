Hi Mike, das ist Bash und nicht Perl, und außerdem nicht sonderlich 
hübsch, aber vielleicht als Anregung...

#!/bin/bash
PSOUTPUT=$(ps --no-headers -C $(basename $0))
COUNT=$(echo -e "$PSOUTPUT" | wc -l)
if [ "$COUNT" -gt "1" ] ; then
         echo "Already running."
         exit 1
else
         echo "Running..."
         sleep 30
         echo "Done."
fi
